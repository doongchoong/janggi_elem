Config

    let div0 = document.getElementById('jb0');

    let board0 = JanggiBoard(div0, {
        position: 'start',
        orientation: 'han',
    });

JanggiBoard(element, config) 함수를 이용해서 원하는 엘리먼트에 장기판을 붙일수 있습니다. config을 이용해서 기본 설정을 할 수 있습니다. 

{
    width: '360px',
    height: '400px',
    orientation: 'cho',    // han
    piece_style: 'janggi', //  chess
    animation: false,
    ilegalmove: false,
    moveable: false,
    position: '',
}

기본값은 위와 같으며 다르게 설정하고 싶은 필드만 설정하면 됩니다.

Methods

// 해당 위치의 기물이 이동할 수 있는 지점들
board.get_dests(position);
// 보드 사이즈 변경
board.resize();
// 시작방향 세팅 (초,한)
board.set_orientation(direction); // 'cho' or 'han'
// 보드 뒤집기
board.flip();
// 보드 초기화 (기물정리, 턴 및 수, 히스토리)
board.clear();
// FEN 파싱 
board.parse_fen(fen_string);
// 시작 포지션 세팅
board.set_position(position_string);
// 기물 이동 
board.move(from,to);
// 이전
board.prev();
// 다음
board.next();
// 한수쉼
board.pass_turn();

사용할 수 있는 메서드 목록

Flip Board

let jb1btn = document.getElementById('jb1btn');
jb1btn.addEventListener('click', function(){
    let jb1 = document.getElementById('jb1');
    let board = JanggiBoard(jb1, {});
    board.flip();
});

flip()를 이용하면 장기판을 뒤집을 수 있습니다.
